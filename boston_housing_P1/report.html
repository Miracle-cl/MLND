def performance_metric2(y_true, y_predict):
    """"""
    
    y_mean = np.array( y_true ).mean()
    
    SS_total = SS_residuals = 0.0
    for i in range( len(y_true) ):
        SS_total += ( (y_true[i] - y_mean)**2 )
        SS_residuals += ( (y_true[i] - y_predict[i])**2 )
    
    score = 1.0 - ( SS_residuals / SS_total )

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    num = len(X)
    num_cv = num / cv
    scores = []
    for i in range(cv):
        X_test = X[i*num_cv:(i+1)*num_cv]
        y_test = y[i*num_cv:(i+1)*num_cv]
        # 
        X_train = X.drop( list(X_test.index) )
        y_train = y.drop( list(X_test.index) )
        estimator.fit( X_train, y_train )
        y_pred = estimator.predict( X_test )
        score = performance_metric( y_test, y_pred )
        scores.append(score)
    return scores



def fit_model2(X,y):

        avg_scores = np.array(scores).mean()
        if avg_scores > maxscore:
            maxscore = avg_scores
